# Configuración para MongoDB local en el mismo VPS
MONGODB_URI=mongodb://localhost:27017/mini-crm
NEXT_PUBLIC_BASE_URL=http://localhost:3000

# OPCIONES ALTERNATIVAS:

# Si MongoDB tiene autenticación:
# MONGODB_URI=mongodb://usuario:password@localhost:27017/mini-crm

# Si MongoDB está en otro puerto:
# MONGODB_URI=mongodb://localhost:27018/mini-crm

# Si usas MongoDB Atlas (en la nube):
# MONGODB_URI=mongodb+srv://usuario:password@cluster.mongodb.net/mini-crm?retryWrites=true&w=majority

# IMPORTANTE:
# - No uses comillas en los valores
# - El nombre de la base de datos es "mini-crm" (puedes cambiarlo si quieres)
# - En producción, actualiza NEXT_PUBLIC_BASE_URL con tu dominio real (ej: https://tu-dominio.com)
